#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define LENGTH 11

int main(void) {
	srand(time(NULL));

	int intArray[LENGTH];

	for (int i = 0; i < LENGTH; i++) {
		intArray[i] = rand() % 11;

		for (int j = 0; j < i; j++) {
			if (intArray[j] == intArray[i]) {
				i--;
			}
		}
	}

	for (int i = 0; i < LENGTH; i++) {
		printf("%d\n", intArray[i]);
	}

	return 0;
}
